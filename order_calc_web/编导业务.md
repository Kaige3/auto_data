## 🚀 VAMS 数据中台：RPA 接口技术文档 (v3.1 - 自动识别版)

### 1. 接口定义 (影刀 -> 后端)
影刀不再需要判断视频来源，直接发送原始 `raw_data` 数组即可。

* **URL:** `POST /api/v1/video/upload`
* **JSON Payload:**
```json
{
  "raw_data": [
    "很难不爱简单慵懒的穿搭#秋冬穿搭 #慵懒风 #卫裤", 
    "13.3万",                                       
    "直播引流、商品广告", // 或者 "12\n1\n3\n2"
    "7.12 d@n.qR... https://v.douyin.com/xxx/...", 
    "发布时间：2026-01-09 14:56"                    
  ]
}
```

---

## 2. 服务端自动判定逻辑

后端在接收到请求后，通过对 `Index 2` 的内容进行“指纹识别”：

| Index 2 内容特征 | 自动识别来源 (source_type) | 存储逻辑 |
| :--- | :--- | :--- |
| **纯数字组** (如 `12\n1...`) | **主页视频** | 拆分为互动数（点赞/评论/收藏等） |
| **文本标签** (如 `直播引流`) | **本地视频** | 拆分为业务属性（BIZ Tags） |

---

## 3. 后端 Python (Flask) 实现 Demo

这个脚本演示了如何实现这种“自动感应”逻辑并写入 SQLite。

    # 播放量转换 (13.3万 -> 133000)

def upload_video():
    data = request.json
    if not data or 'raw_data' not in data:
        return jsonify({"code": 400, "message": "Invalid data"}), 400
    
    try:
        local_id = parse_and_save(data['raw_data'])
        return jsonify({"code": 200, "message": "success", "data": {"id": local_id}})
    except Exception as e:
        return jsonify({"code": 500, "message": str(e)}), 500

DB_PATH = "vams_data.db"

class FeishuSyncer:
    def __init__(self, config):
        self.config = config
        self.tenant_access_token = ""
        self.expire_time = 0

    async def get_token(self, session):
        """获取或刷新飞书 tenant_access_token"""
        if time.time() < self.expire_time - 60:
            return self.tenant_access_token

        url = "https://open.feishu.cn/open-apis/auth/v3/tenant_access_token/internal"
        payload = {
            "app_id": self.config["app_id"],
            "app_secret": self.config["app_secret"]
        }
        async with session.post(url, json=payload) as resp:
            res = await resp.json()
            if res.get("code") == 0:
                self.tenant_access_token = res["tenant_access_token"]
                self.expire_time = time.time() + res["expire"]
                return self.tenant_access_token
            raise Exception(f"获取Token失败: {res.get('msg')}")

    def get_unsynced_data(self):
        """从 SQLite 提取未同步数据，并关联属性表中的标签"""
        conn = sqlite3.connect(DB_PATH)
        conn.row_factory = sqlite3.Row
        cursor = conn.cursor()
        
        # 聚合查询：将 video_attributes 表中的标签聚合成一个字符串
        query = """
            SELECT v.*, 
                   (SELECT GROUP_CONCAT(attr_value) FROM video_attributes 
                    WHERE video_id = v.id AND attr_type = 'BIZ') as biz_tags
            FROM video_records v
            WHERE v.sync_feishu = 0
        """
        rows = cursor.execute(query).fetchall()
        conn.close()
        return rows

    async def batch_sync(self, session, batch_rows):
        """批量推送数据到飞书并执行回写"""
        token = await self.get_token(session)
        url = f"https://open.feishu.cn/open-apis/bitable/v1/apps/{self.config['app_token']}/tables/{self.config['table_id']}/records/batch_create"
        
        headers = {"Authorization": f"Bearer {token}", "Content-Type": "application/json"}
        
        # 构造飞书要求的 fields 结构
        records = []
        for row in batch_rows:
            records.append({
                "fields": {
                    "视频标题": row["title"],
                    "播放量": row["play_count"],
                    "点赞数": row["digg_count"],
                    "视频链接": row["share_url"],
                    "发布时间": int(time.mktime(time.strptime(row["publish_time"], "%Y-%m-%d %H:%M"))) * 1000,
                    "业务标签": row["biz_tags"] if row["biz_tags"] else "",
                    "视频来源": row["video_source"]
                }
            })

        async with session.post(url, headers=headers, json={"records": records}) as resp:
            res = await resp.json()
            if res.get("code") == 0:
                # 提取飞书返回的 record_id 进行回写
                record_ids = [r["record_id"] for r in res["data"]["records"]]
                self.backfill_ids(batch_rows, record_ids)
                return len(record_ids)
            else:
                print(f"批量同步失败: {res.get('msg')}")
                return 0

    def backfill_ids(self, original_rows, feishu_ids):
        """将飞书 Record ID 回写到 SQLite 并标记为同步成功"""
        conn = sqlite3.connect(DB_PATH)
        cursor = conn.cursor()
        for i, row in enumerate(original_rows):
            cursor.execute("""
                UPDATE video_records 
                SET sync_feishu = 1, feishu_record_id = ? 
                WHERE id = ?
            """, (feishu_ids[i], row["id"]))
        conn.commit()
        conn.close()

    async def run(self):
        """主循环：每 10 秒扫描一次数据库"""
        async with aiohttp.ClientSession() as session:
            while True:
                rows = self.get_unsynced_data()
                if rows:
                    print(f"发现 {len(rows)} 条待同步数据...")
                    # 飞书批量接口限制每次最多 100 条
                    for i in range(0, len(rows), 100):
                        chunk = rows[i:i+100]
                        count = await self.batch_sync(session, chunk)
                        print(f"成功同步 {count} 条到飞书。")
                else:
                    print("暂无待同步数据，休眠中...")
                await asyncio.sleep(10)
